function treeSum(arr) {
  let sum = 0;

  for (let s = 0; s < arr.length; s++) {
    let element = arr[s];

    if (Array.isArray(element)) {
      sum = sum + treeSum(element);
    } else if (typeof element === "number") {
      sum = sum + element;
    }
  }
  return sum;
}
